  let responses=Array(40).fill(null);
  let review=Array(40).fill(false);

  function loadQ(){
    const q=questions[current];
    document.getElementById('questionBox').innerText=(current+1)+'. '+q.q;
    const box=document.getElementById('optionsBox');
    box.innerHTML='';
    q.o.forEach((op,i)=>{
      const l=document.createElement('label');
      l.innerHTML=`<input type='radio' name='opt' ${responses[current]===i?'checked':''} onchange='responses[${current}]=${i};updatePalette()'> ${op}`;
      box.appendChild(l);
    });
    updatePalette();
  }

  function buildPalette(){
    const p=document.getElementById('palette');
    p.innerHTML='';
    questions.forEach((_,i)=>{
      const b=document.createElement('button');
      b.innerText=i+1;
      b.onclick=()=>{current=i;loadQ();};
      p.appendChild(b);
    });
    updatePalette();
  }

  function updatePalette(){
    document.querySelectorAll('.qgrid button').forEach((b,i)=>{
      if(review[i]) b.className='marked';
      else if(responses[i]!==null) b.className='attempted';
      else b.className='notattempted';
    });
  }

  function nextQ(){if(current<39){current++;loadQ();}}
  function prevQ(){if(current>0){current--;loadQ();}}

  function submitExam(){
    let score=0;
    responses.forEach((r,i)=>{if(r===questions[i].a)score++;});
    alert('Exam Submitted! Score: '+score+' / 40');
  }

    time--;
  },1000);
